# CURDATE, CURTIME, and NOW 

 SELECT CURTIME();
+-----------+
| CURTIME() |
+-----------+
| 08:16:05  |
+-----------+

SELECT CURDATE();
+------------+
| CURDATE()  |
+------------+
| 2024-04-02 |
+------------+

SELECT NOW();
+---------------------+
| NOW()               |
+---------------------+
| 2024-04-02 08:21:57 |
+---------------------+
